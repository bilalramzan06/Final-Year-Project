int V; // Number of vertices

list *adj;

public:

this->V = V;

adj = new list[V];

}

void addEdge(int v, int w) {

adj[v].push_back(w);

}

};

function fib(n) {

if (n <= 1) return n;

return fib(n - 1) + fib(n - 2);

}

let memo = {};

function fibMemo(n) {

if (n in memo) return memo[n];

memo[n] = fib(n);

return memo[n];

}
